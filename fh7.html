Write a C++ program to append new data to an existing text file:

            #include < iostream.h >
#include < fstream.h >
#include < string.h >

int main() {
    // Open the text file for appending
    std::ofstream outFile("existing_file.txt", std::ios::app);

    // Check if the file is opened successfully
    if (!outFile) {
        std::cerr << "Failed to open the file.\n";
        return 1;
    }

    // Append new data to the file
    std::string newData = "This is new data appended to the existing file.\n";
    outFile << newData;

    // Close the file
    outFile.close();

    std::cout << "New data appended to the file successfully.\n";

    return 0;
}

Output:
